CREATE DATABASE student_system;
CREATE TABLE stus(
	id SERIAL PRIMARY KEY,
	name VARCHAR(50),
	course VARCHAR(40),
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
	
CREATE TABLE student_logs(
	log_id SERIAL PRIMARY KEY,
	student_id INT,
	old_name VARCHAR(40),
	old_course VARCHAR(50),
	action VARCHAR(40),
	action_time TIMESTAMP
);

CREATE FUNCTION log_student_update()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
	INSERT INTO student_logs(student_id,old_name,old_course,action)
	VALUES(OLD.id,OLD.name,OLD.course,'UPDATE');
	RETURN NEW;
	END;
	$$;

CREATE TRIGGER trg_student_update
BEFORE UPDATE ON stus
FOR EACH ROW
EXECUTE FUNCTION log_student_update();

CREATE OR REPLACE FUNCTION get_student_count(course_name TEXT)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE 
    total INT;
BEGIN
    SELECT COUNT(*) 
    INTO total 
    FROM stus 
    WHERE course = course_name;

    RETURN total;
END;
$$;

AS $$
BEGIN
    INSERT INTO student_logs(
        student_id,
        old_name,

        new_course,
        action
    )
    VALUES (
        OLD.id,
        OLD.name,
        OLD.course,
        NEW.name,
        NEW.course,
        'UPDATE'
    );

    RETURN NEW;
END;
$$;

CREATE OR REPLACE PROCEDURE manage_student(
	p_id INT,
	P_name TEXT,
	p_course TEXT
)
LANGUAGE plpgsql
AS $$
BEGIN
	IF EXISTS ( SELECT 1 FROM stus WHERE id =p_id)THEN
		UPDATE stus
	SET name = p_name,
	course = p_course
		WHERE id =p_id;
	ELSE
		INSERT INTO stus(id,name,course)
		VALUES (p_id,p_name,p_course);
	END IF;
END;
$$;

CREATE OR REPLACE PROCEDURE update_multiple_courses()
LANGUAGE plpgsql
AS $$
DECLARE
    r RECORD;
BEGIN
    FOR r IN SELECT id, course FROM stus LOOP
        CALL manage_student(r.id, r.id || '_updated', r.course || '_new');
    END LOOP;
END;
$$;
