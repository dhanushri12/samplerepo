CREATE FUNCTION get_bonus(salary NUMERIC)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
	BEGIN
		RETURN salary * 0.10;
	END;
	$$;

CREATE TABLE employees_logs(
	log_id SERIAL PRIMARY KEY,
	emp_id INT,
	old_salary NUMERIC,
	action_time TIMESTAMP 
	DEFAULT CURRENT_TIMESTAMP);

CREATE 	FUNCTION log_salary_change()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
	BEGIN
		INSERT INTO employees_logs(emp_id,old_salary)
		VALUES(OLD.id,OLD.salary);
		RETURN NEW;
	END;
	$$;
	
CREATE TRIGGER try_salary_change
BEFORE UPDATE ON employees
FOR EACH ROW
EXECUTE FUNCTION log_salary_change();

CREATE OR REPLACE FUNCTION check_temp()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
	BEGIN
		IF NEW.temperature > 50 THEN 
		INSERT INTO alarms(sensor_id,temperature)
		VALUES (NEW.sensor_id,NEW.temperature);
	END IF;
RETURN NEW;
	END;
	$$;

CREATE TRIGGER temp_alert
AFTER INSERT ON readings
FOR EACH ROW
EXECUTE FUNCTION check_temp();
